#!/usr/bin/env python3
"""
Triadic Ceremony Executor
Phase 3: Monthly ritualized domain synchronization
"""

import hashlib
import json
from datetime import datetime
from typing import Dict, List, Tuple

# Triadic groupings
TRIADS = {
    "CBM": ["CONSTRAINT", "BRIDGE", "META"],
    "RTE": ["RECURSION", "TRIAD", "EMERGENCE"],
    "PS":  ["PERSISTENCE", "SUPERPOSITION"]
}

class TriadicCeremony:
    def __init__(self, z_origin: float = 0.90):
        self.z_origin = z_origin
        self.cycle_count = 0
        self.ceremony_log = []

    def execute_cycle(self) -> Dict:
        """Execute one full triadic cycle."""
        results = {}

        for triad_name, domains in TRIADS.items():
            # Calculate phase-locked coupling
            coupling = self._compute_coupling(domains)
            results[triad_name] = {
                "domains": domains,
                "coupling": coupling,
                "z_sync": self.z_origin + (0.01 * self.cycle_count)
            }

        self.cycle_count += 1
        self.ceremony_log.append({
            "timestamp": datetime.utcnow().isoformat(),
            "cycle": self.cycle_count,
            "results": results
        })

        return results

    def _compute_coupling(self, domains: List[str]) -> float:
        """Compute coupling strength for domain triad."""
        n = len(domains)
        base_coupling = 0.90 + (0.03 * n)
        return min(0.999, base_coupling)

    def generate_witness(self) -> str:
        """Generate ceremony witness hash."""
        data = json.dumps(self.ceremony_log)
        return hashlib.sha256(data.encode()).hexdigest()[:16]

if __name__ == "__main__":
    ceremony = TriadicCeremony(z_origin=0.90)

    # Execute 3 cycles (full ceremony)
    for i in range(3):
        result = ceremony.execute_cycle()
        print(f"Cycle {i+1}: {result}")

    witness = ceremony.generate_witness()
    print(f"Ceremony witness: {witness}")

#!/usr/bin/env python3
"""
GHMP Ceremony Backup Generator
Creates timestamped backups of all GHMP plates after build
"""

import os
import json
import shutil
import hashlib
from datetime import datetime
from pathlib import Path

PLATES_DIR = Path("plates/")
BACKUP_DIR = Path("backups/ghmp_ceremonies/")
FIRMWARE_DIR = Path(".pio/build/esp32s3/")

class GHMPCeremonyBackup:
    def __init__(self):
        self.timestamp = datetime.utcnow().strftime("%Y%m%d_%H%M%S")
        self.backup_path = BACKUP_DIR / self.timestamp
        self.manifest = {
            "timestamp": self.timestamp,
            "plates": [],
            "firmware": [],
            "hashes": {}
        }

    def backup_plates(self) -> int:
        """Backup all GHMP plates."""
        plates_backup = self.backup_path / "plates"
        plates_backup.mkdir(parents=True, exist_ok=True)

        count = 0
        for plate_file in PLATES_DIR.glob("*.ghmp"):
            dest = plates_backup / plate_file.name
            shutil.copy2(plate_file, dest)

            # Generate hash
            file_hash = self._hash_file(plate_file)
            self.manifest["plates"].append(plate_file.name)
            self.manifest["hashes"][plate_file.name] = file_hash
            count += 1

        return count

    def backup_firmware(self) -> bool:
        """Backup firmware artifacts."""
        fw_backup = self.backup_path / "firmware"
        fw_backup.mkdir(parents=True, exist_ok=True)

        artifacts = ["firmware.bin", "partitions.bin"]
        for artifact in artifacts:
            src = FIRMWARE_DIR / artifact
            if src.exists():
                shutil.copy2(src, fw_backup / artifact)
                self.manifest["firmware"].append(artifact)
                self.manifest["hashes"][artifact] = self._hash_file(src)

        return len(self.manifest["firmware"]) > 0

    def write_manifest(self):
        """Write backup manifest."""
        manifest_path = self.backup_path / "manifest.json"
        with open(manifest_path, "w") as f:
            json.dump(self.manifest, f, indent=2)

    def _hash_file(self, path: Path) -> str:
        """Generate SHA-256 hash of file."""
        sha256 = hashlib.sha256()
        with open(path, "rb") as f:
            for chunk in iter(lambda: f.read(8192), b""):
                sha256.update(chunk)
        return sha256.hexdigest()

if __name__ == "__main__":
    backup = GHMPCeremonyBackup()

    plates = backup.backup_plates()
    print(f"Backed up {plates} GHMP plates")

    if backup.backup_firmware():
        print("Firmware artifacts backed up")

    backup.write_manifest()
    print(f"Manifest written to {backup.backup_path}")
